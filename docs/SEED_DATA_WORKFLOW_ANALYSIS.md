```bash
psql $DATABASE_URL -f supabase/seed.sql

psql $DATABASE_URL -f supabase/seed-part3.sql
```

### Verification Queries

```sql
-- Verify all users created
SELECT COUNT(*) FROM profiles; -- Should be 11

-- Verify role assignments
SELECT 
    p.first_name, 
    p.last_name, 
    r.name as role,
    r.level
FROM user_roles ur
JOIN profiles p ON ur.user_id = p.id
JOIN roles r ON ur.role_id = r.id
ORDER BY r.level;

-- Verify task hierarchy
SELECT 
    t.title,
    creator.first_name || ' ' || creator.last_name as created_by,
    assignee.first_name || ' ' || assignee.last_name as assigned_to
FROM tasks t
JOIN profiles creator ON t.created_by = creator.id
JOIN profiles assignee ON t.assigned_to = assignee.id
ORDER BY t.created_at;

-- Verify budget tracking
SELECT 
    b.name,
    b.total_amount,
    b.spent_amount,
    ROUND((b.spent_amount / b.total_amount * 100)::numeric, 1) as percent_spent
FROM budgets b;
```
